function initFirebase() {
  try {
    if (typeof firebase === 'undefined') {
      // SEGURANÇA: SDK não carregado — mantém tela de login visível, nunca concede acesso
      console.error('[WORKRAIL] Firebase SDK não carregado. Acesso bloqueado.');
      showToast('Erro: Firebase indisponível. Verifique sua conexão e recarregue a página.', 'error');
      // login-overlay permanece visível (estado padrão) — nenhum acesso concedido
      return;
    }
    if (FIREBASE_CONFIG.apiKey === 'SUA_API_KEY_AQUI') {
      // Modo desenvolvimento: Firebase não configurado — acesso demo com privilégio mínimo
      console.warn('[WORKRAIL] Firebase não configurado. Modo desenvolvimento ativo (somente leitura).');
      mostrarAppSemLogin('vendas'); // privilégio mínimo em modo dev, nunca super_admin
      return;
    }
    firebase.initializeApp(FIREBASE_CONFIG);
    db      = firebase.firestore();
    storage = firebase.storage();
    auth    = firebase.auth();
    firebaseOk = true;
    console.log('[WORKRAIL] Firebase inicializado com sucesso.');

    // ── Observa estado de autenticação ──────────────────────
    auth.onAuthStateChanged(async (user) => {
      if (user) {
        await carregarPerfilUsuario(user);
      } else {
        currentUser = null;
        document.getElementById('login-overlay').classList.remove('hidden');
      }
    });

  } catch(e) {
    // SEGURANÇA: falha na inicialização — mantém tela de login visível, nunca concede acesso
    console.error('[WORKRAIL] Erro crítico ao inicializar Firebase:', e);
    showToast('Erro ao conectar com o servidor. Tente recarregar a página.', 'error');
    // login-overlay permanece visível — nenhum acesso concedido
  }
}

MUDANÇA CRÍTICA: Removeu mostrarAppSemLogin('super_admin') em casos de erro
ANTES: SDK ausente ou erro → concedia super_admin silenciosamente (⚠️ RISCO)
DEPOIS: SDK ausente ou erro → mostra erro e mantém tela de login (✅ SEGURO)
DEV MODE: apiKey placeholder → usa 'vendas' (privilégio mínimo) em vez de 'super_admin'

TESTE: Desabilitar Firebase SDK → deve mostrar erro, não dar acesso
TESTE: Configurar apiKey como 'SUA_API_KEY_AQUI' → deve entrar como vendas

async function confirmAprovar() {
  if (!validateStep_Gestor('aprovar')) return;
  closeMod('mAprovar');
  showLoading(true);
  try {
    const obs = document.querySelector('#screen-gestor textarea')?.value?.trim();

    // Upload obrigatório: checklist de instalação
    const uploadsChecklist = await uploadMultiplos(
      [['fChecklist','CHECKLIST_INSTALACAO']], 'checklists'
    );

    // Recuperar o Nº do Pedido gerado na abertura da solicitação
    const numPedido = currentSolicitacao?.dadosAdm?.numeroPedido
                   || document.getElementById('numeroPedidoSolic')?.value
                   || document.getElementById('campoPedidoNum')?.value
                   || '';

    const hist = entradaHistorico('APROVADO_GESTOR','Gestor','gestor',obs);
    await atualizarSolicitacao(currentSolicitacaoId,
      { 'parecerGestor.decisao':'APROVADO', 'parecerGestor.observacoes':obs, 'dadosAdm.numeroPedido': numPedido },
      STATUS.APROVADO_GESTOR.key, 'adm', hist
    );

    // Registrar checklist no histórico de documentos
    if (uploadsChecklist.length > 0) {
      await adicionarDocumentos(currentSolicitacaoId,
        uploadsChecklist.map(u => ({ ...u, etapa:'gestor', responsavel: currentUser?.nome || 'Gestor' })));
    }

    const prot = currentSolicitacao?.protocolo;
    await enviarNotificacao(EMAIL_RECIPIENTS.adm,
      'Solicitação Aprovada — Revisão Contratual',
      `A solicitação foi aprovada pelo Gestor e aguarda revisão contratual.`,
      prot, { Observacoes: obs, Cliente: currentSolicitacao?.dadosCliente?.nomeCliente }
    );
    await enviarNotificacao(FORMSUBMIT_EMAIL,
      'Solicitação Aprovada pelo Gestor', `Obs: ${obs}`, prot);

    pendingFiles = {};
    showToast('Solicitação aprovada e encaminhada ao ADM de Vendas!', 'success');
    go('adm');
  } catch(e) {
    console.error(e);
    showToast('Erro ao aprovar solicitação.', 'error');
  } finally {
    showLoading(false);
  }
}

MUDANÇA: 
  1. Adicionado uploadMultiplos para o checklist (fChecklist)
  2. Adicionado adicionarDocumentos para registrar no Firebase
  3. Pasta de destino: solicitacoes/checklists/

FLUXO: 
  Validação ✓ → Upload ✓ → Atualizar Status ✓ → Registrar Docs ✓ → Email ✓ → ADM

TESTE: Aprovar solicitação com checklist → verificar em Firebase Storage

function buildSbHistory(){
  const el=document.getElementById('sbHistList');
  if(!el) return;
  const recent=HISTORY.slice(0,4);
  if(recent.length===0){
    el.innerHTML=`<div style="text-align:center;padding:16px 8px;color:var(--text-muted);font-size:12px">Nenhuma solicitação registrada.</div>`;
    return;
  }
  el.innerHTML=recent.map(h=>`
    <div class="sb-hist-item" onclick="goPage('historico')">
      <div class="sh-row">
        <div class="sh-code">${h.code.replace(/WR-\d{4}-/,'#')}</div>
        <div class="sh-chip chip ${h.status}" style="font-size:9px;padding:1px 5px">${h.statusTxt}</div>
      </div>
      <div class="sh-name">${h.client}</div>
    </div>`).join('');
}
